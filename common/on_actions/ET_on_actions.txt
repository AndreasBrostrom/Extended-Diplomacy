on_actions = {

	on_startup = {
		effect = {
			set_variable = { et_var_conscription = 0.00 }
		}
	}

	on_monthly = {
		effect = {
			if = { limit = { is_debug = yes } log = "[GetDateText]: [This.Owner.GetTag] on_monthly trigger!" }
			if = {
				limit = { has_idea = scraping_the_barrel }
				set_variable = { et_var_conscription = -0.050 }
			}
			else_if = {
				limit = { has_idea = all_adults_serve }
				set_variable = { et_var_conscription = -0.030 }
			}
			else_if = {
				limit = { has_idea = service_by_requirement }
				set_variable = { et_var_conscription = -0.020 }
			}
			else_if = {
				limit = { has_idea = extensive_conscription }
				set_variable = { et_var_conscription = -0.010 }
			}
			else_if = {
				limit = { has_idea = limited_conscription }
				set_variable = { et_var_conscription = -0.005 }
			}
			else_if = {
				limit = { has_idea = volunteer_only }
				set_variable = { et_var_conscription = -0.005 }
			}
			else_if = {
				limit = { has_idea = disarmed_nation }
				set_variable = { et_var_conscription = -0.005 }
			}
			else = {
				set_variable = { et_var_conscription = 0.000 }
			}
		}
	}

	#ROOT - country that just went to peace
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					NOT = { has_global_flag = ET_focus_global_newera_1_flag }
					any_country = {
						has_war = no
					}
					date > 1945.01.01
				}
				set_global_flag = ET_focus_global_newera_1_flag
			}
			FROM = {
				set_temp_variable = { ET_threat_neg = 0 }
				subtract_from_temp_variable = { ET_threat_neg = has_added_tension_amount }
				
				add_named_threat = {
					threat = ET_threat_neg
					name = ET_lost_war
				}
			}
		}
	}

	#used when puppeting through the occupied territories menu during peace time (or when releasing from non-core but owned territory, f.e. Britain releasing Egypt)
	#ROOT = nation being released, FROM = overlord
	on_release_as_puppet = {
		effect = {
			if = {
				limit = { has_global_flag = ET_focus_global_newera_1_flag }
				ROOT = { ET_set_new_era_effect = yes }
			}
		}
	}

	#ROOT is free nation FROM is releaser
	on_release_as_free = {
		effect = {
			if = {
				limit = { has_global_flag = ET_focus_global_newera_1_flag }
				ROOT = { ET_set_new_era_effect = yes }
			}
		}
	}
}
